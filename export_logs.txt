<script>
function openDrawer(el) {
  var id = el.dataset.id;
  var text = el.dataset.task;
  var name = el.dataset.username;
  var time = el.dataset.time;
  var dept = el.dataset.department;
  var project = el.dataset.project;
  var isKb = el.dataset.iskb === 'true';
  var color = el.dataset.color;

  // Avatar
  var av = document.getElementById('drawer-avatar');
  av.textContent = name.substring(0,2).toUpperCase();
  av.style.background = color;

  document.getElementById('drawer-name').textContent = name;
  document.getElementById('drawer-sub').textContent  =
    time + (dept ? ' · ' + dept : '');

  // Badges
  var badges = document.getElementById('drawer-badges');
  badges.innerHTML = '';
  if (project) {
    badges.innerHTML += '<span style="background:var(--primary-subtle);color:var(--primary);font-size:.72rem;font-weight:600;padding:.2rem .6rem;border-radius:5px">' + project + '</span>';
  }
  if (isKb) {
    badges.innerHTML += '<span style="background:#d1fae5;color:#065f46;font-size:.72rem;font-weight:600;padding:.2rem .6rem;border-radius:5px"><i class="fas fa-lightbulb"></i> Knowledge Base</span>';
  } else {
    badges.innerHTML += '<span style="background:#e0f2fe;color:#0369a1;font-size:.72rem;font-weight:600;padding:.2rem .6rem;border-radius:5px">Task / Issue</span>';
  }

  document.getElementById('drawer-text').textContent = text;

  document.getElementById('al-drawer').classList.add('open');
  document.getElementById('al-drawer-overlay').classList.add('open');
}

function closeDrawer() {
  document.getElementById('al-drawer').classList.remove('open');
  document.getElementById('al-drawer-overlay').classList.remove('open');
}
</script>
